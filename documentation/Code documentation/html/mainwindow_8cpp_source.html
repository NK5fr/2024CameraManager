503 //check if folder contains project subfolders ie. output,input,images
504 bool isOkProjectFolderPath(const QString& folderPath){
505  if (folderPath.isNull()) return false;
506 
507  QDir dir(folderPath);
508  QStringList fileList = dir.entryList();
509  bool hasInputFolder = false;
510  bool hasOutputFolder = false;
511  bool hasImagesFolder = false;
512 
513  foreach(QString item, fileList ){
514  if (item.contains("input")) hasInputFolder = true;
515  if (item.contains("output")) hasOutputFolder = true;
516  if (item.contains("images")) hasImagesFolder = true;
517  if (item.contains("cal_run")) return true; //open anyway
518  }
519  return hasInputFolder && hasOutputFolder && hasImagesFolder;
520 }
