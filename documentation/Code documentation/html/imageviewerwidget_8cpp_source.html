181 char* ImageViewerWidget::unpackPGMFile(const QString& filepath, int64_t* const bufferSize, int64_t* const imageWidth, int64_t* const imageHeight) {
182  QFile file(filepath);
183  if (!file.open(QIODevice::ReadOnly)) return nullptr;
184  QByteArray blob = file.readAll();
185  const char* data = blob.constData();
186  file.close();
187  const int tempBufferSize = 128;
188  char tempBuffer[tempBufferSize];
189  int startIndex = 0;
190  int element = 0;
191  unsigned int width = 0;
192  unsigned int height = 0;
193  char* imageData = nullptr;
194  unsigned int imageSize = 0;
195  for (int i = 0; i < blob.size(); i++) {
196  char c = data[i];
197  if (element == 4) {
198  startIndex = i;
199  break;
200  }
201  if (c == '\0') break;
202  if (c >= 'a' && c <= 'z' || c >= 'A' && c <= 'Z' || c >= '0' && c <= '9') {
203  if (i - startIndex >= tempBufferSize - 1) break;
204  tempBuffer[i - startIndex] = c;
205  tempBuffer[(i - startIndex) + 1] = '\0';
206  } else {
207  //if (startIndex - i <= 0) continue;
208  QString label(tempBuffer);
209  if (element == 0) {
210  if (!label.contains("P5")) return nullptr;
211  } else if (element == 1) {
212  width = label.toUInt();
213  } else if (element == 2) {
214  height = label.toUInt();
215  }
216  element++;
217  startIndex = i + 1;
218  }
219  }
220  if (element < 4) {
221  *imageWidth = imageWidget.getTexture()->getTextureWidth();
222  *imageHeight = imageWidget.getTexture()->getTextureHeight();
223  imageSize = imageWidget.getTexture()->getTextureSize();
224  *bufferSize = imageSize;
225  imageData = new char[imageSize];
226  memset(imageData, 0, imageSize);
227  return imageData;
228  }
229  imageSize = blob.size() - startIndex;
230  *bufferSize = imageSize;
231  *imageWidth = width;
232  *imageHeight = height;
233  imageData = new char[imageSize];
234  memcpy(imageData, data + startIndex, imageSize);
235  return imageData;
