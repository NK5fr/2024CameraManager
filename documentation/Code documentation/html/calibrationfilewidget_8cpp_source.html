160 void CalibrationFileWidget::updateCombinationTable() {
161  combinationList->clear();
162  vector<TrackPoint::CameraCombination*> camComb = calibFile->getCameraCombinations();
163  for (int i = 0; i < camComb.size(); i++) {
164  QString combination = QString::number(camComb[i]->camNumbers[0]) + "_" + QString::number(camComb[i]->camNumbers[1]) + "_" + QString::number(camComb[i]->camNumbers[2]);
165  QTreeWidgetItem* item = new QTreeWidgetItem();
166  item->setText(0, combination);
167  if (camComb[i]->textStatus.isEmpty()) {
168  item->setText(1, QString::number(camComb[i]->s0));
169  } else {
170  item->setText(1, camComb[i]->textStatus);
171  }
172  item->setText(2, QString::number(camComb[i]->numFrames));
173  item->setData(0, Qt::UserRole, QVariant::fromValue((quintptr) camComb[i]));
174  updateFiltersCamerasOnly(camComb[i]);
175  if (camComb[i]->status == TrackPoint::CalibrationStatus::Warning) {
176  item->setBackgroundColor(0, warningColor);
177  } else if (camComb[i]->status == TrackPoint::CalibrationStatus::Failed) {
178  item->setBackgroundColor(0, failedColor);
179  } else {
180  item->setBackgroundColor(0, okColor);
181  }
182  bool warningS0 = (camComb[i]->s0 > ((QLineEdit*) filterList->itemWidget(filterList->invisibleRootItem()->child(0), 1))->text().toDouble());
183  bool failedS0 = (camComb[i]->s0 > ((QLineEdit*) filterList->itemWidget(filterList->invisibleRootItem()->child(0), 2))->text().toDouble());
184  if (warningS0) item->setBackgroundColor(1, warningColor);
185  if (failedS0 || !camComb[i]->textStatus.isEmpty()) item->setBackgroundColor(1, failedColor);
186  else if (!warningS0 && !failedS0) item->setBackgroundColor(1, okColor);
187  if (camComb[i]->numFrames < ((QLineEdit*) filterList->itemWidget(filterList->invisibleRootItem()->child(4), 1))->text().toDouble()) {
188  item->setBackgroundColor(2, failedColor);
189  } else {
190  item->setBackgroundColor(2, okColor);
191  }
192  combinationList->addTopLevelItem(item);
193  }
194  combinationList->resizeColumnToContents(0);
195  combinationList->resizeColumnToContents(1);
196  combinationList->resizeColumnToContents(2);

207 void CalibrationFileWidget::updateCameraTable() {
208  cameraTable->clear();
209  vector<TrackPoint::CameraCombination*> camComb = calibFile->getCameraCombinations();
210  vector<TrackPoint::Camera*> cams = calibFile->getCameras();
211  for (int i = 0; i < cams.size(); i++) {
212  int ok = 0;
213  int warning = 0;
214  int failed = 0;
215  for (int j = 0; j < camComb.size(); j++) {
216  if (cams[i]->camNo != camComb[j]->camNumbers[0] && cams[i]->camNo != camComb[j]->camNumbers[1] && cams[i]->camNo != camComb[j]->camNumbers[2]) {
217  failed++;
218  continue;
219  }
220  if (camComb[j]->status == TrackPoint::OK) {
221  ok++;
222  } else if (camComb[j]->status == TrackPoint::Warning) {
223  warning++;
224  } else if (camComb[j]->status == TrackPoint::Failed) {
225  failed++;
226  }
227  }
228  QTreeWidgetItem* item = new QTreeWidgetItem();
229  item->setText(0, QString::number(cams[i]->camNo));
230  item->setText(1, QString::number(cams[i]->serialNo));
231  item->setText(2, QString::number(ok));
232  item->setText(3, QString::number(warning));
233  item->setText(4, QString::number(failed) + "/" + QString::number(camComb.size()));
234  cameraTable->addTopLevelItem(item);
235  }
236  cameraTable->resizeColumnToContents(0);
237  cameraTable->resizeColumnToContents(1);
238 }
