101 vector<Vector3d*> SocketViewerWidget::readLine(QString& line) {
102  QRegularExpression coordsRegEx("(-?\\d+(?:\\.?\\d*[eE]?\\-?\\d*)?)(?:\\s*)(-?\\d+(?:\\.?\\d*[eE]?\\-?\\d*)?)(?:\\s*)(-?\\d+(?:\\.?\\d*[eE]?\\-?\\d*)?)");
103  QRegularExpressionMatchIterator i = coordsRegEx.globalMatch(line);
104  vector<Vector3d*> vectors;
105  while (i.hasNext()) {
106  QRegularExpressionMatch match = i.next();
107  Vector3d* v = new Vector3d();
108  v->x = match.captured(1).toDouble();
109  v->y = match.captured(3).toDouble();
110  v->z = match.captured(2).toDouble();
111  vectors.push_back(v);
112  }
113  return vectors;
114 }

522 void HostAddressDialog::displayError(QAbstractSocket::SocketError socketError) {
523  switch (socketError) {
524  case QAbstractSocket::RemoteHostClosedError:
525  QMessageBox::information(this, "Camera Manager Client",
526  "Remote Host Closed.");
527  break;
528  case QAbstractSocket::HostNotFoundError:
529  QMessageBox::information(this, "Camera Manager Client",
530  "The host was not found. Please check the "
531  "host name and port settings.");
532  break;
533  case QAbstractSocket::ConnectionRefusedError:
534  QMessageBox::information(this, "Camera Manager Client",
535  "The connection was refused by the peer. "

537  "and check that the host name and port "
538  "settings are correct.");
539  break;
540  default:
541  QMessageBox::information(this, "Camera Manager Client", QString("The following error occurred: %1.").arg(tcpSocket->errorString()));
